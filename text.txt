    sidebar.style.display = "block";

    content.classList.add("full-width");

    content.innerHTML = `<h2>${type.charAt(0).toUpperCase()+type.slice(1)} Grading</h2>

function loadContent(section, el) {
    document.querySelectorAll(".sidebar a").forEach(a => a.classList.remove("active"));
    if (el) el.classList.add("active");

    if (section === "home") {
        content.innerHTML = `
        <h2>School Profile</h2>
        <p>Mission, Vision, and Core Values of SDIHS.</p>`;
    }

    if (section === "shs") {
        content.innerHTML = `
        <h2>SHS</h2>
        <p>HUMSS, ICT, and EPAS strands offered.</p>`;
    }

    if (section === "jhs") {
        content.innerHTML = `

    }
}
